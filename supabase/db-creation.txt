create table rest_areas (
  id bigint generated by default as identity primary key,
  inserted_at timestamp with time zone default timezone('utc'::text, now()) not null,
  updated_at timestamp with time zone default timezone('utc'::text, now()) not null,
  name text,
  reference text,
  region text,
  surface text,
  creator text,
  bbq boolean,
  fuel boolean,
  lights boolean,
  showers boolean,
  toilets boolean,
  tables boolean,
  power boolean,
  wifi boolean,
  ligths boolean,
  ligthing_type text,
  mobile_reception boolean,
  length integer,
  width integer,
  dangerous_bays integer default 0,
  dangerous_bays_available integer default 0,
  truck_bays integer default 0,
  truck_bays_available integer default 0,
  images text,
  cover_image text,
  longitude text,
  latitude text
);

SELECT AddGeometryColumn ('','rest_areas','geom',4326,'POINT',2);
